from typing import List, Dict

def evaluate(rules: List[Dict], applicant_data: Dict) -> Dict:
    triggered_rules = []
    for rule in rules:
        if all(rule.get('conditions', {}).get(key, False) == value for key, value in applicant_data.items()):
            triggered_rules.append({
                'rule_id': rule['id'],
                'reason_ids': rule.get('reason_ids', []),
                'decision': rule.get('decision', 'approve')
            })
    
    # Sort by rule priority ascending
    triggered_rules.sort(key=lambda x: rules.index(x))
    
    overall_decision = 'approve'
    for rule in triggered_rules:
        if rule['decision'] == 'reject':
            overall_decision = 'reject'
            break
        elif rule['decision'] == 'refer':
            overall_decision = 'refer'
            break
    
    return {
        'overall_decision': overall_decision,
        'triggered_rules': triggered_rules
    }
"""

from app.rules import evaluate

def test_multi_rule_case():
    applicant_data = {
        "income": "60k",
        "credit_score": "550",
        "employment_status": "employed"
    }
    
    result = evaluate([
